Inserting and Unloading a Linked List

Nodes

typedef struct node
{
    char *phrase;
    struct node *next;
}
node;

Creating a Linked List

node *list = NULL;
node *n = malloc(sizeof(node));
if (n == NULL)
{
    return 1;
}
n->phrase = "Hi!";
n->next = NULL;
list = n;

Inserting Nodes

n = malloc(sizeof(node));
n->phrase = "Hey!";
n->next = list;
list = n;

// image

free(list);

node *ptr = list->next;
free(list);
list = ptr;
ptr = list->next;
free(list);
list = ptr;
