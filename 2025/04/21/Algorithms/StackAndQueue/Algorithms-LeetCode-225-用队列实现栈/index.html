LeetCode 225. 用队列实现栈

单个队列模拟栈

class MyStack {
public:

    queue<int> que;

    MyStack() {
        
    }
    
    void push(int x) {
        que.push(x);
    }
    
    int pop() {
        int size = que.size() - 1;
        while (size--) {
            que.push(que.front());
            que.pop();
        }

        int a = que.front();
        que.pop();
        return a;
    }
    
    int top() {
        int a = this->pop();
        this->push(a);
        return a;
    }
    
    bool empty() {
        return que.empty();
    }
};
